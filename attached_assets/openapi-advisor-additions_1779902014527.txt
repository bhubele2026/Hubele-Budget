# === Step 2: Add these path entries under the existing paths: section   ===

  /advisor/nudge:
    get:
      operationId: getAdvisorNudge
      tags: [advisor]
      summary: Get a single proactive financial observation for the dashboard
      description: |
        Returns a cached (1h TTL) AI-generated observation about the
        household's current financial state. The frontend renders this
        as a dismissible card on the Dashboard. When the advisor is
        disabled (no API key, ADVISOR_ENABLED=false, or no meaningful
        data yet), returns `enabled: false` and the UI hides itself.
      responses:
        "200":
          description: Nudge (possibly disabled)
          content:
            application/json:
              schema: { $ref: "#/components/schemas/AdvisorNudge" }
        "401":
          description: Unauthorized

  /advisor/chat:
    post:
      operationId: postAdvisorChat
      tags: [advisor]
      summary: Send a message to the AI budget advisor
      description: |
        Stateless chat endpoint. The client passes the full conversation
        history each turn; the server tacks on the latest message and
        calls the model. The server augments the system prompt with a
        live snapshot of the household's budget, cashflow, and debts.
      requestBody:
        required: true
        content:
          application/json:
            schema: { $ref: "#/components/schemas/AdvisorChatRequest" }
      responses:
        "200":
          description: Advisor reply
          content:
            application/json:
              schema: { $ref: "#/components/schemas/AdvisorChatResponse" }
        "400":
          description: Invalid request body
        "401":
          description: Unauthorized
        "503":
          description: Advisor not configured on this server

# ========================================================================
# === Step 3: Add these schemas under components.schemas:                ===
# ========================================================================

    AdvisorNudge:
      type: object
      additionalProperties: false
      required: [enabled]
      properties:
        enabled:
          type: boolean
          description: When false, the frontend hides the nudge UI entirely.
        severity:
          type: string
          enum: [info, warn, alert]
          description: |
            UI hint for color/icon. `info` = neutral observation, `warn`
            = something to keep an eye on, `alert` = something likely to
            cause a problem if ignored.
        message:
          type: string
          description: 1–2 sentence observation in plain English.
        source:
          type: string
          enum: [advisor, empty]
          description: |
            `advisor` = generated by the model. `empty` = the model
            decided there was nothing worth surfacing right now (the UI
            should hide).
        generatedAt:
          type: string
          format: date-time

    AdvisorChatMessage:
      type: object
      additionalProperties: false
      required: [role, content]
      properties:
        role:
          type: string
          enum: [user, assistant]
        content:
          type: string

    AdvisorChatRequest:
      type: object
      additionalProperties: false
      required: [message]
      properties:
        message:
          type: string
          minLength: 1
          maxLength: 4000
        history:
          type: array
          maxItems: 24
          items: { $ref: "#/components/schemas/AdvisorChatMessage" }

    AdvisorChatResponse:
      type: object
      additionalProperties: false
      required: [message]
      properties:
        message:
          type: string
        usage:
          type: object
          additionalProperties: false
          properties:
            inputTokens:
              type: integer
            outputTokens:
              type: integer
