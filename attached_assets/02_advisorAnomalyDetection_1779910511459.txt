// Anomaly detection for the AI advisor.
//
// Two tools that surface unusual activity by comparing what's happening
// now against the household's own historical baselines:
//
// 1. detect_spending_anomalies — flags categories whose current pace is
//    statistically far from their 6-month rolling average. Uses a simple
//    z-score-style threshold so spikes get caught without false alarms
//    on naturally volatile categories.
//
// 2. detect_unusual_transactions — flags individual transactions in the
//    last 30 days whose amount is far outside the typical range for
//    their category (or for the household if uncategorized).
//
// Read-only. No new tables — all baselines are computed on demand from
// existing transaction history. Cheap enough to call mid-conversation.
//
// Also exports computeAnomalies() and computeUnusualTransactions() as
// plain functions so the weekly insights / proactive nudge can reuse
// them without going through the tool layer.

import { z } from "zod";
import { and, desc, eq, gte, lt, sql } from "drizzle-orm";
import {
  db,
  budgetCategoriesTable,
  transactionsTable,
} from "@workspace/db";
import { registerTool } from "./advisorTools";
import { buildHistoryBlock, type CategoryTrend } from "./advisorHistoryContext";

// ---------------------------------------------------------------------------
// Spending anomalies (category-level)
// ---------------------------------------------------------------------------

export interface SpendingAnomaly {
  category: string;
  avgSpend6mo: number;
  currentMtd: number;
  paceProjection: number;
  paceDeltaPct: number;
  zScore: number | null;
  severity: "info" | "warn" | "alert";
  reason: string;
}

// Tuning knobs. Conservative defaults: surface 30%+ deviations as info,
// 60%+ as warn, 100%+ (with a non-trivial dollar amount) as alert. Z-score
// gates filter out categories whose 6mo std-dev makes the deviation look
// statistically noisy.
const DEFAULT_MIN_DOLLAR_AMOUNT = 25;
const DEFAULT_INFO_PCT = 30;
const DEFAULT_WARN_PCT = 60;
const DEFAULT_ALERT_PCT = 100;
const DEFAULT_MIN_HISTORY_MONTHS = 3;

export function computeAnomaliesFromTrends(
  trends: CategoryTrend[],
  opts: {
    minDollarAmount?: number;
    infoPct?: number;
    warnPct?: number;
    alertPct?: number;
  } = {},
): SpendingAnomaly[] {
  const minDollar = opts.minDollarAmount ?? DEFAULT_MIN_DOLLAR_AMOUNT;
  const infoPct = opts.infoPct ?? DEFAULT_INFO_PCT;
  const warnPct = opts.warnPct ?? DEFAULT_WARN_PCT;
  const alertPct = opts.alertPct ?? DEFAULT_ALERT_PCT;

  const out: SpendingAnomaly[] = [];
  for (const t of trends) {
    // Need a real historical baseline AND a non-trivial pace.
    if (t.avgSpend6mo < minDollar) continue;
    if (t.paceProjection < minDollar) continue;
    if (t.paceDeltaPct == null) continue;

    const absDelta = Math.abs(t.paceDeltaPct);
    if (absDelta < infoPct) continue;

    // Z-score against historical std-dev — gates out categories whose
    // 6mo history is naturally lumpy (e.g. once-a-quarter expenses).
    const zScore = t.stdDev6mo > 0
      ? (t.paceProjection - t.avgSpend6mo) / t.stdDev6mo
      : null;
    if (zScore != null && Math.abs(zScore) < 1.0) continue;

    let severity: SpendingAnomaly["severity"] = "info";
    if (absDelta >= alertPct) severity = "alert";
    else if (absDelta >= warnPct) severity = "warn";

    const direction = t.paceDeltaPct > 0 ? "above" : "below";
    const reason = `Current pace $${t.paceProjection.toFixed(2)} is ${absDelta}% ${direction} the 6-month average of $${t.avgSpend6mo.toFixed(2)}.`;

    out.push({
      category: t.category,
      avgSpend6mo: t.avgSpend6mo,
      currentMtd: t.currentMtd,
      paceProjection: t.paceProjection,
      paceDeltaPct: t.paceDeltaPct,
      zScore: zScore != null ? Math.round(zScore * 100) / 100 : null,
      severity,
      reason,
    });
  }

  // Sort by severity (alert > warn > info), then by absolute delta.
  const sevWeight = { alert: 3, warn: 2, info: 1 };
  out.sort((a, b) => {
    const sw = sevWeight[b.severity] - sevWeight[a.severity];
    if (sw !== 0) return sw;
    return Math.abs(b.paceDeltaPct) - Math.abs(a.paceDeltaPct);
  });
  return out;
}

export async function computeAnomalies(
  householdId: string,
  opts: {
    minDollarAmount?: number;
    infoPct?: number;
    warnPct?: number;
    alertPct?: number;
  } = {},
): Promise<SpendingAnomaly[]> {
  const history = await buildHistoryBlock(householdId);
  // Need at least N months of history for category to be eligible.
  const minHistoryMonths = DEFAULT_MIN_HISTORY_MONTHS;
  const trends = history.categoryTrends.filter((t) => {
    // Heuristic: stdDev=0 with avg>0 means single-month data; skip.
    if (t.avgSpend6mo > 0 && t.stdDev6mo === 0) return false;
    return true;
  });
  void minHistoryMonths;
  return computeAnomaliesFromTrends(trends, opts);
}

// ---------------------------------------------------------------------------
// Unusual transactions (row-level)
// ---------------------------------------------------------------------------

export interface UnusualTransaction {
  transactionId: string;
  date: string;
  description: string;
  amount: number;
  category: string | null;
  categoryMedian: number;
  zScore: number | null;
  reason: string;
}

export async function computeUnusualTransactions(
  householdId: string,
  opts: { lookbackDays?: number; minAmount?: number } = {},
): Promise<UnusualTransaction[]> {
  const lookbackDays = opts.lookbackDays ?? 30;
  const minAmount = opts.minAmount ?? 50;
  const now = new Date();
  const lookbackStart = new Date(now);
  lookbackStart.setDate(lookbackStart.getDate() - lookbackDays);
  const baselineStart = new Date(now);
  baselineStart.setDate(baselineStart.getDate() - 180);

  function iso(d: Date): string {
    return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, "0")}-${String(
      d.getDate(),
    ).padStart(2, "0")}`;
  }

  // Pull the recent transactions (candidates).
  const candidates = await db
    .select({
      id: transactionsTable.id,
      date: transactionsTable.occurredOn,
      description: transactionsTable.description,
      amount: transactionsTable.amount,
      categoryId: transactionsTable.categoryId,
      categoryName: budgetCategoriesTable.name,
    })
    .from(transactionsTable)
    .leftJoin(
      budgetCategoriesTable,
      eq(transactionsTable.categoryId, budgetCategoriesTable.id),
    )
    .where(
      and(
        eq(transactionsTable.householdId, householdId),
        gte(transactionsTable.occurredOn, iso(lookbackStart)),
        sql`${transactionsTable.amount} < 0`,
        sql`abs(${transactionsTable.amount}) >= ${minAmount}`,
      ),
    )
    .orderBy(desc(transactionsTable.occurredOn));

  if (candidates.length === 0) return [];

  // Pull baseline distribution per category from the prior 6 months
  // (excluding the lookback window so a recent spike doesn't drag the
  // baseline up and mask itself).
  const baseline = await db
    .select({
      categoryId: transactionsTable.categoryId,
      amount: transactionsTable.amount,
    })
    .from(transactionsTable)
    .where(
      and(
        eq(transactionsTable.householdId, householdId),
        gte(transactionsTable.occurredOn, iso(baselineStart)),
        lt(transactionsTable.occurredOn, iso(lookbackStart)),
        sql`${transactionsTable.amount} < 0`,
      ),
    );

  // Build per-category amount arrays (absolute values).
  const byCat = new Map<string, number[]>();
  for (const row of baseline) {
    const key = row.categoryId ?? "__uncat";
    const arr = byCat.get(key) ?? [];
    arr.push(Math.abs(Number(row.amount)));
    byCat.set(key, arr);
  }

  function quantile(sortedArr: number[], q: number): number {
    if (sortedArr.length === 0) return 0;
    const pos = (sortedArr.length - 1) * q;
    const base = Math.floor(pos);
    const rest = pos - base;
    if (sortedArr[base + 1] !== undefined) {
      return sortedArr[base] + rest * (sortedArr[base + 1] - sortedArr[base]);
    }
    return sortedArr[base];
  }

  function stddev(values: number[], mean: number): number {
    if (values.length === 0) return 0;
    const v = values.reduce((s, x) => s + (x - mean) ** 2, 0) / values.length;
    return Math.sqrt(v);
  }

  const stats = new Map<string, { median: number; p90: number; mean: number; sd: number; n: number }>();
  for (const [k, arr] of byCat.entries()) {
    const sorted = [...arr].sort((a, b) => a - b);
    const mean = arr.reduce((s, v) => s + v, 0) / arr.length;
    stats.set(k, {
      median: quantile(sorted, 0.5),
      p90: quantile(sorted, 0.9),
      mean,
      sd: stddev(arr, mean),
      n: arr.length,
    });
  }

  const out: UnusualTransaction[] = [];
  for (const c of candidates) {
    const abs = Math.abs(Number(c.amount));
    const key = c.categoryId ?? "__uncat";
    const s = stats.get(key);
    if (!s || s.n < 5) continue; // need enough baseline data
    if (abs <= s.p90) continue;
    const z = s.sd > 0 ? (abs - s.mean) / s.sd : null;
    // Require both: above p90 AND z >= 2 (when sd available)
    if (z != null && z < 2) continue;
    out.push({
      transactionId: c.id,
      date: c.date,
      description: c.description,
      amount: Number(c.amount),
      category: c.categoryName,
      categoryMedian: Math.round(s.median * 100) / 100,
      zScore: z != null ? Math.round(z * 100) / 100 : null,
      reason: `$${abs.toFixed(2)} is ${z != null ? z.toFixed(1) + "σ" : "above the 90th percentile"} for ${c.categoryName ?? "uncategorized"} (typical: $${s.median.toFixed(2)}).`,
    });
  }
  // Top of list = biggest absolute outliers first.
  out.sort((a, b) => Math.abs(b.amount) - Math.abs(a.amount));
  return out;
}

// Tools
// ---------------------------------------------------------------------------

const detectAnomaliesInput = z.object({
  minDollarAmount: z
    .number()
    .min(0)
    .optional()
    .describe("Floor below which a category is ignored. Default 25."),
  infoPct: z
    .number()
    .min(1)
    .optional()
    .describe("Threshold for 'info' severity (% deviation). Default 30."),
  warnPct: z
    .number()
    .min(1)
    .optional()
    .describe("Threshold for 'warn' severity (% deviation). Default 60."),
  alertPct: z
    .number()
    .min(1)
    .optional()
    .describe("Threshold for 'alert' severity (% deviation). Default 100."),
});

registerTool({
  name: "detect_spending_anomalies",
  description:
    "Find categories where current-month pace is unusually high or low vs the household's own 6-month baseline. Filters out naturally noisy categories via z-score. Use when the user asks 'where am I overspending' in a baseline-aware way, or 'is anything weird this month'.",
  riskTier: "read",
  inputSchema: detectAnomaliesInput,
  jsonSchema: {
    type: "object",
    properties: {
      minDollarAmount: { type: "number", minimum: 0 },
      infoPct: { type: "number", minimum: 1 },
      warnPct: { type: "number", minimum: 1 },
      alertPct: { type: "number", minimum: 1 },
    },
    additionalProperties: false,
  },
  handler: async (input, ctx) => {
    const anomalies = await computeAnomalies(ctx.householdId, input);
    return {
      result: {
        count: anomalies.length,
        anomalies,
      },
    };
  },
});

const detectUnusualTxnsInput = z.object({
  lookbackDays: z
    .number()
    .int()
    .min(1)
    .max(180)
    .optional()
    .describe("How many recent days to scan. Default 30."),
  minAmount: z
    .number()
    .min(0)
    .optional()
    .describe("Ignore transactions whose absolute value is below this. Default 50."),
});

registerTool({
  name: "detect_unusual_transactions",
  description:
    "Find individual transactions in the lookback window whose amount is statistically unusual for their category (above 90th percentile AND >=2 std devs from category mean). Uses the household's own 6 months of history as the baseline. Use to surface single big transactions worth a second look.",
  riskTier: "read",
  inputSchema: detectUnusualTxnsInput,
  jsonSchema: {
    type: "object",
    properties: {
      lookbackDays: { type: "integer", minimum: 1, maximum: 180 },
      minAmount: { type: "number", minimum: 0 },
    },
    additionalProperties: false,
  },
  handler: async (input, ctx) => {
    const unusual = await computeUnusualTransactions(ctx.householdId, input);
    return {
      result: {
        count: unusual.length,
        unusual,
      },
    };
  },
});
